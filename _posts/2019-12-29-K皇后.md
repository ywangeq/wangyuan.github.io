>经典的K皇后问题，四楼都是递归回溯
> 我们用一个一维数组来表达K皇后的某一个解，然后使用递归看是否可以通过

```
def Kqueen(n):
    board = [-1 for i in range(n)]
    res=[]
    def check(k,j):#判断第k个皇后能不能放在第j列
        for i in range(k):
            if board[i]==j or abs(board[i]-j)==abs(k-i):
                return False
        return true
    def dfs(depth,value):
        if depth==n: 
            res.append(value)
            return
        for i in range(n):
            if check(depth,i):
                board[depth]=i
                s='.'*n
                dfs(depth+1,value+[s[:i]+'Q'+s[1+i:]])
    dfs(0,[])
    return res
```
