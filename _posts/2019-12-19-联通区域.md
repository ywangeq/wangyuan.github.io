> 给定一个二维数组（0，1），求数组中最多有多少个1连通,只考虑上下左右

- follow up 也有求多少个连通区域 

```
def numIs(grid):
    res=[]
    if len(grid)==0: return 0
    def dfs(i,j,co,go):
        if grid[i][j]==1:
            co+=1
            grid[i][j]=0
            dfs(i+1,j,co,go)
            dfs(i,j+1,co,go)
            dfs(i-1,j,co,go)
            dfs(i,j-1,co,go)
            go = max(co,go)
            res.append(go)
    co，go=1,0
    step=0
    grid =[[0] + row + [0] for row in grid]
    grid =[[0]*len(grid[0])]+ grid + [[0]*len(grid[0])]
    for i in range(1,len(grid)-1):
        for j in range(1,len(grid[0])-1):
            if grid[i][j]==1:
                step+=1
            dfs(i,j,co,go)
    return step

```
