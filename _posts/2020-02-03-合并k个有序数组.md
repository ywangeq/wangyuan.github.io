class Solution:
    def mergeSort(self,nums):
        if len(nums)<=1: return nums
        mid = len(nums)//2
        left = self.mergeSort(nums[:mid])
        right = self.mergeSort(nums[mid:])
        return Sort(left[0],right[0])
    def Sort(self,left,right):
        res =[]
        a,b =0,0
        while a<len(left) and b<len(right):
            if left[a]<right[b]:
                res.append(left[a])
                a+=1
            else:
                res.append(right[b])
                b+=1
        while a < len(left):
            res.append(left[a])
            a+=1
        while b <len(right):
            res.append(right[b])
            b_-1
        return [res]
